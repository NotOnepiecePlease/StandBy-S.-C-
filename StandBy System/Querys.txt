DECLARE @cont as int = 0
DECLARE @ordem_serv as int = 0

WHILE @cont <= 33986  
BEGIN 
	if (select count(sv_id) from tb_servicos where sv_id = @cont) > 0
		begin
			UPDATE tb_servicos  
			SET sv_ordem_serv = @ordem_serv
			WHERE sv_id = @cont

			PRINT @cont
			SET @cont = @cont + 1
			SET @ordem_serv = @ordem_serv + 1
		end
	else
		begin
			SET @cont = @cont + 1
		end
END  








CREATE TRIGGER TGR_VENDAS_AI
ON tb_servicos
FOR INSERT
AS
BEGIN
    DECLARE
    @OrdemServico  int

    SELECT @OrdemServico = sv_ordem_serv FROM INSERTED
    UPDATE tb_log SET log_ultima_ordem_servico = @OrdemServico
END
GO
